    async function carregarDados() {
      const response = await fetch('https://cronometregemgps-default-rtdb.firebaseio.com/.json');
      const dados = await response.json();
      const corpoTabela = document.querySelector('#tabela-resultados tbody');
      corpoTabela.innerHTML = '';

      const resultados = Object.values(dados).map(piloto => ({
        nome: piloto.nome || 'Desconhecido',
        kart: piloto.kart || '',
        melhor_volta: piloto.melhor_volta || '--:--:---',
        tempo_atual: piloto.tempo_atual || '--:--:---',
        voltas: piloto.voltas || 0
      }));

      resultados.sort((a, b) => {
        if (!a.melhor_volta || !b.melhor_volta) return 0;
        return a.melhor_volta.localeCompare(b.melhor_volta);
      });

      resultados.forEach((p, i) => {
        const tr = document.createElement('tr');
        if (i === 0) tr.classList.add('highlight');
        tr.innerHTML = `
          <td>${p.nome}</td>
          <td>${p.kart}</td>
          <td>${p.melhor_volta}</td>
          <td>${p.tempo_atual}</td>
          <td>${p.voltas}</td>
        `;
        corpoTabela.appendChild(tr);
      });
    }

    setInterval(carregarDados, 2000);
    carregarDados();
  </script></body>
